** defining it
#+begin_src haskell
  let hz :: Pattern p => p Note -> p ValueMap
      hz hzPat = n $ fmap f hzPat where
	f freq = log freq / log 2 * 12
#+end_src
** using it
 #+begin_src haskell
   d1 $ s "supersquare"
     |*| slow 4 (hz "1 2 3 4")
     |+| n "1 2 3 4"
 #+end_src
